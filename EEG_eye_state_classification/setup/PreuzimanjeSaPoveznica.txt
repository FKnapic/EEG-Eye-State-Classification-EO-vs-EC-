import os
import subprocess

# Putanja do glavne mape gdje je dataset
base_path = '/Users/filipknapic/Desktop/Diplomski_projekt/data'

# Prolazimo kroz svih 60 ispitanika, samo za prvu sesiju
for sub in range(1, 61):  # 1 do 60
    sub_folder = f'sub-{sub:02d}'
    session_folder = 'ses-session1'  # samo prva sesija
    eeg_folder = os.path.join(base_path, sub_folder, session_folder, 'eeg')

    if os.path.exists(eeg_folder):
        os.chdir(eeg_folder)
        print(f'Enabling remote in: {eeg_folder}')

        try:
            # Aktiviraj udaljeni repozitorij (za git-annex)
            subprocess.run(['git', 'annex', 'enableremote', 'c07eaa42-b8e8-4e8d-a56e-c91f9c08510a'], check=True)
            print(f'Successfully enabled remote in {eeg_folder}')

            # Skini sve datoteke (.eeg, .vhdr, .vmrk)
            subprocess.run(['git', 'annex', 'get', '.'], check=True)
            print(f'Successfully retrieved files in {eeg_folder}\n')

        except subprocess.CalledProcessError as e:
            print(f'Failed to enable remote or retrieve files in {eeg_folder}: {e}\n')
    else:
        print(f'Folder does not exist: {eeg_folder}')
